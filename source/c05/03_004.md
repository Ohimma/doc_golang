#### 2. go

```
package main

import "fmt"

func main() {
	data := []int{31, 30, 43, 38, 15, 17}
	quickSort(data)
}

func quickSort(arr []int) {
	fmt.Printf("插入排序前 arr=%v\n", arr)
	for i := 1; i < len(arr); i++ {
		for j := i; j > 0; j-- {
			fmt.Printf("开始比较 j=%v < j-1=%v\n", arr[j], arr[j-1])
			if arr[j] < arr[j-1] {
				arr[j], arr[j-1] = arr[j-1], arr[j]
			} else {
				break
			}
		}
		fmt.Println("插入排序后 arr=", arr)
	}
}
```

```
// 方法一 双 for
Array.prototype.quickSort = function() {
    for (let i = 1; i < this.length; i++) {
        for (let j = i; j > 0; j--) {
            console.log("开始比较 j=",this[j], " < j-1=", this[j-1])
            if (this[j] < this[j-1]) {
                const tmp = this[j]
                this[j] = this[j-1]
                this[j-1] = tmp
            } else {
                break
            }
        }
        console.log("插入排序后 arr=",this)
    }
};

// 方法二  this指针 + while
Array.prototype.quickSort = function() {
    for (let i = 1; i < this.length; i++) {
        let j = i;
        while (j) {
          // 如果选择的这个值要小，就将前面的大值后移
          console.log("开始比较 j=",this[j], " < j-1=", this[j-1])
          if (this[j] < this[j - 1]) {
            const tmp = this[j]
            this[j] = this[j - 1];
            this[j-1] = tmp
          } else {
            break;
          }
          j--;
        }
        console.log("插入排序后 arr=",this)
    }
};

var arr = [31, 30, 43, 38, 15, 17]
```
