## 2.005 双指针: 找无重复最长子串

#### 1. 需求

```
给定一个字符串，请你找出其中不含有重复字符的 最长子串 的长度。

输入: s = "abcabcbb"
输出: 3
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

输入: s = "bbbbb"
输出: 1
解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

输入: s = "pwwkew"
输出: 3
解释: 因为无重复字符的最长子串是 "wke"，所以其长度为 3。
     请注意，你的答案必须是 子串 的长度，"pwke" 是一个子序列，不是子串。

输入: s = ""
输出: 0

```

package main

import (
    "fmt"
)
func main() {
   s := "abcabcdabcabc"
   result := lengthOfLongestSubstring(s)
   fmt.Println("result = ", result)
}
func lengthOfLongestSubstring(s string) int {
        freq := make([]int, 128)
        var res = 0
        start,end := 0,-1
        for start<len(s){
                fmt.Println("start = ", start)
                if end+1<len(s)&&freq[s[end+1]] == 0{
                        end++
                        freq[s[end]]++
                        fmt.Println("end+1 = ", freq)
                }else{
                        freq[s[start]]--
                        start++
                        fmt.Println("start+1 = ", &freq)
                }
                res = max(res, end-start+1)
        }
        return res
}
func max(i,j int)int{
        if i>j{
                return i
        }else{
                return j
        }
}

var findSubstr = function (s) {
  let l = 0;       // 窗口左指针
  let maxLen = 0;  // 需要返回的最大长度
  const m = new Map(); // 记录每次的出现的值

  for (let r = 0; r < s.length; r++) {
    const c = s[r];
    if (m.has(c) && m.get(c) >= l) {
      // 左指针移动到出现的重复值的下标 + 1处，即更新窗口，去除重复值
      // 如果出现重复的值,并且值在当前窗口内,左指针右移一位
      l = m.get(c) + 1;
    }
    // 更新最大的子串长度
    maxLen = Math.max(maxLen, r - l + 1);
    console.log("c = ", c , " l = ", l, " maxLen = ", maxLen)
    // 存入字典表中：[字符，下标]
    m.set(c, r);
    console.log("m = ", m)
  }
  return maxLen;
};

var s = "abcabcdabcabc"
console.log(findSubstr(s))
