package main

import (
	"fmt"
	_ "math/rand"
)

func main() {
	data := []int{31, 30, 43, 38, 15, 17}
	//for i := 0; i < 5; i++ {
	//	data = append(data, rand.Intn(100))
	//}
	quickSort(data)
}

func quickSort(arr []int) {
	n := len(arr)
	fmt.Printf("选择排序前 arr=%v\n", arr)
	for i := 0; i < n-1; i++ {
		for j := i + 1; j < n; j++ {
			fmt.Printf("开始比较 j=%v < i=%v\n", arr[j], arr[i])
			if arr[j] < arr[i] {
				arr[i], arr[j] = arr[j], arr[i]
			}
		}
		fmt.Printf("选择排序后 arr=%v\n", arr)
	}
}
