## 1.2.1 常用包: 命令行参数

#### 1. args 原始写法

```
package main
import (
    "fmt"
    "os"
)

func main() {

    fmt.Println("命令行参数有", len(os.Args))
    //

    for i, v := range os.Args {
        fmt.Printf("args[%v]=%v\n", i, v)
    }
}
```

#### 2. flag 指定参数形式

<https://golang.org/pkg/flag/#pkg-index>

flag.String() 方法，返回指针类型  
flag.StringVar() 方法，返回值类型

```
package main

import (
  "flag"
  "fmt"
)
 
func main() {
    // String(参数选项, 默认值, 参数说明)
    username := flag.String("name", "", "Input your username")
    flag.Parse()
    fmt.Println("Hello, ", *username)
}

$ go run .\flag.go -name=world
>>> Hello, world
```

```
package main
import (
    "fmt"
    "flag"
)

func main() {

    var user string
    var pwd string
    var host string
    var port int

    // StringVar(指定参数值, 参数选项, 默认值, 参数说明)
    flag.StringVar(&user, "u", "", "用户名，默认为空")
    flag.StringVar(&pwd,  "pwd", "", "密码, 默认为空")
    flag.StringVar(&host, "h", "127.0.0.1", "连接地址, 默认127.0.0.1")
    flag.IntVar(&port, "port", 3306, "端口号, 默认3306")

    // 非常重要的参数，转换
    flag.Parse()

    fmt.Printf("user=%v pwd=%v host=%v port=%v\n",
               user, pwd, host, port)
}

$ go run .\test.go -u root -pwd cdscsd#@3 -h 127.0.0.2
>>> user=root pwd=cdscsd#@3 host=127.0.0.2 port=3306
```
